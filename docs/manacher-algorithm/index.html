let SubPalindromes = function (s) {
  let n = s.length;
  let newS = "^#";
  for (let i = 0; i < n; i++) {
    newS += s[i] + "#";
  }
  newS += "$";
  let p = new Array(newS.length).fill(0);
  let center = 0,
    r = 0;
  for (let i = 1; i < newS.length - 1; i++) {
    let mirror = 2 * center - i;

      p[i]++;

  }
